1. Ve a supabase.com → Crea cuenta gratis
2. Crea un nuevo proyecto (elige región más cercana)
3. Ve a SQL Editor y ejecuta este comando:
CREATE TABLE pronosticos (
  id BIGSERIAL PRIMARY KEY,
  nombre TEXT NOT NULL,
  cargo TEXT,
  campeon TEXT,
  goleador TEXT,
  p1 TEXT, p2 TEXT, p3 TEXT,
  final TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE pronosticos ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON pronosticos FOR ALL USING (true) WITH CHECK (true);
